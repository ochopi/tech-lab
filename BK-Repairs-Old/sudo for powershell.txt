# Sudo - Ejecuta comandos con privilegios elevados
function sudo {
    <#
    .EXAMPLE
    sudo Get-Service
    sudo ipconfig /flushdns -Close
    
    .SYNOPSIS
    Ejecuta comandos con privilegios de administrador.
    
    .DESCRIPTION
    Si la sesión actual ya está elevada, ejecuta el comando inline.
    Si no, intenta elevar usando (en orden): Windows Terminal, pwsh.exe o powershell.exe.
    Por defecto, la ventana elevada permanece abierta. Usa -Close para cerrarla automáticamente.
    
    .PARAMETER Command
    El comando a ejecutar.
    
    .PARAMETER Close
    Si se especifica, cierra la ventana elevada al terminar el comando.
    #>

    [CmdletBinding()]
    param(
        [Parameter(ValueFromRemainingArguments = $true, Mandatory = $true)]
        [string[]]$Command,

        [switch]$Close
    )

    # Detectar si ya estamos elevados
    $isElevated = ([Security.Principal.WindowsPrincipal][Security.Principal.WindowsIdentity]::GetCurrent()
    ).IsInRole([Security.Principal.WindowsBuiltInRole]::Administrator)

    $cmdLine = $Command -join ' '

    # Si ya estamos elevados, ejecutar directamente
    if ($isElevated) {
        try {
            Invoke-Expression $cmdLine
        } catch {
            Write-Error "Error ejecutando el comando: $_"
        }
        return
    }

    # Codificar comando en base64 (UTF-16LE)
    $bytes = [Text.Encoding]::Unicode.GetBytes($cmdLine)
    $b64 = [Convert]::ToBase64String($bytes)

    # Construir argumentos (por defecto con -NoExit, a menos que se use -Close)
    $baseArgs = @('-NoProfile')
    if (-not $Close) { $baseArgs += '-NoExit' }
    $baseArgs += @('-EncodedCommand', $b64)

    # Buscar ejecutables disponibles
    $wt = Get-Command wt.exe -ErrorAction SilentlyContinue
    $pwsh = Get-Command pwsh.exe -ErrorAction SilentlyContinue
    $legacy = Get-Command powershell.exe -ErrorAction SilentlyContinue

    try {
        # Prioridad: Windows Terminal > PowerShell 7 > PowerShell clásico
        if ($wt) {
            $wtArg = "pwsh $($baseArgs -join ' ')"
            Start-Process -FilePath $wt.Source -Verb RunAs -ArgumentList $wtArg
        }
        elseif ($pwsh) {
            Start-Process -FilePath $pwsh.Source -Verb RunAs -ArgumentList $baseArgs
        }
        elseif ($legacy) {
            Start-Process -FilePath $legacy.Source -Verb RunAs -ArgumentList $baseArgs
        }
        else {
            Write-Error "No se encontró wt.exe, pwsh.exe ni powershell.exe. No se puede elevar."
        }
    } catch {
        Write-Error "Error al intentar elevar: $_"
    }
}
